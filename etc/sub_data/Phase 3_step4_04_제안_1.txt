async def get_message_by_tg_id(
    session: AsyncSession,
    *,
    channel_id: int,
    tg_message_id: int,
) -> Message | None:
    return await session.scalar(
        select(Message).where(

async def list_versions(
    session: AsyncSession,
    *,
    message_id: int,
) -> list[MessageVersion]:
    result = await session.scalars(
        select(MessageVersion)
        .where(MessageVersion.message_id == message_id)
        .order_by(MessageVersion.version_no.asc())
    )
    return list(result)


async def list_normalize_jobs(
    session: AsyncSession,
    *,
    target_id: int,
) -> list[AnalysisJob]:
    result = await session.scalars(
        select(AnalysisJob)
        .where(

            AnalysisJob.target_id == target_id,
            AnalysisJob.job_type == JobType.NORMALIZE,
        )
        .order_by(AnalysisJob.id.asc())
    )
    return list(result)


async def count_rows(session: AsyncSession, model) -> int:
    return int(await session.scalar(select(func.count()).select_from(model)) or 0)

# 4) `conftest.py`에 한 가지 추가할 점

앞서 드린 `conftest.py` 초안은 `aware_dt`와 `FakeTelegramMessage`를 일반 함수/클래스로만 두었습니다.
지금처럼 테스트 함수 인자로 직접 쓰려면 **fixture로 노출**하는 편이 더 편합니다.

아래 두 개를 `conftest.py`에 추가하시면 됩니다.

```python
@pytest.fixture
def aware_dt_fixture():
    return aware_dt


@pytest.fixture
def fake_telegram_message_cls():
    return FakeTelegramMessage
```

그런데 pytest에서 이름을 그대로 `aware_dt`, `FakeTelegramMessage`로 fixture로 두면 원래 함수/클래스명과 충돌 느낌이 있을 수 있어서, 개인적으로는 테스트에서는 그냥 import 없이 직접 fixture 이름을 아래처럼 바꾸는 편을 권합니다.

예를 들면:

```python
@pytest.mark.asyncio
async def test_xxx(async_session, channel, fake_telegram_message_cls, aware_dt_fixture):
    tg_msg = fake_telegram_message_cls(
        id=1,
        message="테스트",
        date=aware_dt_fixture("2026-03-22T09:00:00+00:00"),
    )
```

하지만 사용 편의상 기존 이름을 유지하고 싶다면 fixture명을 그대로 써도 됩니다.

# 6) 지금 단계의 핵심 체크포인트

이 정리본 기준으로 가장 먼저 확인할 것은 두 가지입니다.

하나는 `conftest.py`의 공용 fixture가 실제 모델 필드와 맞는지입니다.
특히 `Channel`, `Message`의 필수 컬럼이 프로젝트 모델과 다르면 factory만 조금 수정하면 됩니다.

다른 하나는 `JobType`, `JobStatus`, `JobTargetType`, `ResultType`가 enum인지 문자열인지입니다.
문자열 기반이면 assertion만 문자열로 바꾸면 됩니다.
