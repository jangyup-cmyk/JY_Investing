async def get_job(async_session: AsyncSession, job_id: int) -> AnalysisJob | None:
    return await async_session.get(AnalysisJob, job_id)


async def list_jobs(async_session: AsyncSession) -> list[AnalysisJob]:
    result = await async_session.scalars(
        select(AnalysisJob).order_by(AnalysisJob.id.asc())
    )
    return list(result)


async def seed_job(
    async_session: AsyncSession,
    *,
    job_type: JobType = JobType.NORMALIZE,
    target_type: JobTargetType = JobTargetType.MESSAGE,
    target_id: int = 1,
    status: JobStatus = JobStatus.PENDING,
    priority: int = 100,
    attempt_count: int = 0,
    max_attempts: int = 3,
    scheduled_at: datetime | None = None,
    lease_expires_at: datetime | None = None,
    worker_name: str | None = None,
    idempotency_key: str = "job-key-1",
    payload_json: dict | None = None,
) -> AnalysisJob:
    job = AnalysisJob(
        job_type=job_type,
        target_type=target_type,
        target_id=target_id,
        status=status,
        priority=priority,
        attempt_count=attempt_count,
        max_attempts=max_attempts,
        scheduled_at=scheduled_at or now_utc(),
        started_at=None,
        finished_at=None,
        worker_name=worker_name,
        lease_expires_at=lease_expires_at,
        last_heartbeat_at=None,
        idempotency_key=idempotency_key,
        payload_json=payload_json or {},
    )
    async_session.add(job)
    await async_session.commit()
    await async_session.refresh(job)
    return job

@pytest.mark.asyncio
async def test_enqueue_job_creates_pending_job(
    async_session: AsyncSession,
    message: Message,
):
    job = await AnalysisJobService.enqueue_job(
        async_session,
        job_type=JobType.NORMALIZE,
        target_type=JobTargetType.MESSAGE,
        target_id=message.id,
        idempotency_key=f"normalize:message:{message.id}:version:1:cv1",
        payload_json={"message_id": message.id, "version_id": 1},
        priority=100,
        max_attempts=3,
    )
    await async_session.commit()

    stored = await get_job(async_session, job.id)

    assert stored is not None
    assert stored.status == JobStatus.PENDING
    assert stored.job_type == JobType.NORMALIZE
    assert stored.target_type == JobTargetType.MESSAGE
    assert stored.target_id == message.id
    assert stored.priority == 100
    assert stored.attempt_count == 0
    assert stored.max_attempts == 3

@pytest.mark.asyncio
async def test_claim_next_job_claims_pending_job_by_priority(
    async_session: AsyncSession,
    message: Message,
):
    low_priority = await seed_job(
        async_session,
        job_type=JobType.NORMALIZE,
        target_id=message.id,
        priority=200,
        idempotency_key="job-low",
    )

        async_session,
        job_type=JobType.NORMALIZE,
        target_id=message.id,
        priority=10,
        idempotency_key="job-high",

    claimed = await AnalysisJobService.claim_next_job(
        async_session,
        worker_name="worker-a",
        allowed_job_types=[JobType.NORMALIZE],

    assert stored is not None
    assert stored.status == JobStatus.RUNNING
    assert stored.worker_name == "worker-a"
    assert stored.attempt_count == 1
    assert stored.started_at is not None
    assert stored.lease_expires_at is not None

    still_pending = await get_job(async_session, low_priority.id)
    assert still_pending is not None
    assert still_pending.status == JobStatus.PENDING

추가로 확인할 점이 하나 있습니다.
제가 앞선 워커 초안에서 사용한 `AnalysisJobService.claim_next_job()`는 `updated_at`을 이용한 낙관적 claim 패턴이었기 때문에, 실제 모델에 `updated_at` mixin이 제대로 들어가 있어야 이 테스트와 런타임이 안정적으로 맞습니다. 모델에서 `created_at`, `updated_at`가 빠져 있거나 server default 동작이 다르면 그 부분만 맞춰주시면 됩니다.
